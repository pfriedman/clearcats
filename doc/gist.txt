find duplicate records
  
SELECT DISTINCT t1.id, t1.last_name, t1.first_name, t1.era_commons_username, t2.era_commons_username, t1.netid, t2.netid 
FROM people t1, people t2 
WHERE t1.last_name = t2.last_name AND t1.first_name = t2.first_name AND t1.id <> t2.id
ORDER BY t1.last_name
